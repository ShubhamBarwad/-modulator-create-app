Modules

Module Structure

modules/
  mymodule/
    index.ts
    controller/
    model/
    module.json

Lifecycle Hooks

  • init(app, config) — Setup logic
  • registerRoutes(router) — Register HTTP routes
  • registerModels(orm) — Register database models
  • onShutdown() — Cleanup logic

Registering Routes

// index.ts
export default {
  registerRoutes(app) {
    app.get('/hello', (req, res) => res.send('Hello from my module!'));
  }
};

Registering Events

import { getEventBus } from '@modulator/core';
const eventBus = getEventBus();
eventBus.on('something', handler);
eventBus.emit('something', data);
